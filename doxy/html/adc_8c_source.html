36 //
37 // ATtiny45/85
38 // ===========
39 //
40 // The 10-bit Analog to Digital Converter (ADC) on the ATtinyX5 MCU is used
41 // to provide power and position feedback from the servo circuitry. The
42 // analog inputs are assigned as follows:
43 //
44 // ADC0 (PB5) - Power input
45 // ADC3 (PB3) - Position input
46 //
47 //
48 // ATmega8
49 // =======
50 //
51 // The 10-bit Analog to Digital Converter (ADC) on the ATmega MCU is used
52 // to provide power and position feedback from the servo circuitry. The
53 // analog inputs are assigned as follows:
54 //
55 // ADC0 (PC0) - Power input
56 // ADC1 (PC1) - Voltage input
57 // ADC2 (PC2) - Position input
58 //
59 // NOTE: Unlike the ATtiny45/85 and ATmega168 the ATmega8 ADC sampling
60 // cannot be directly started from a timer compare. Therefore, we have
61 // the timer interrupt kick off an ADC sample.
62 //
63 //
64 // ATmega168
65 // =========
66 //
67 // The 10-bit Analog to Digital Converter (ADC) on the ATmega MCU is used
68 // to provide power and position feedback from the servo circuitry. The
69 // analog inputs are assigned as follows:
70 //
71 // ADC0 (PC0) - Power input
72 // ADC1 (PC1) - Voltage input
73 // ADC2 (PC2) - Position input
74 //
75 
76 // Defines for the power and position channels.
77 #define ADC_CHANNEL_POWER 0
78 #define ADC_CHANNEL_POSITION 1
79 #define ADC_CHANNEL_VOLTAGE 2
80 
81 // The ADC clock prescaler of 64 is selected to yield a 125 KHz ADC clock
82 // from an 8 MHz system clock.
83 #define ADPS ((1<<ADPS2) | (1<<ADPS1) | (0<<ADPS0))
84 
85 // The timer clock prescaler of 1024 is selected to yield a 7.8125 KHz ADC clock
86 // from an 8 MHz system clock.
87 #define CSPS ((1<<CS02) | (0<<CS01) | (1<<CS00))
88 
89 // Define the compare register value to generate a timer interrupt and initiate
90 // an ADC sample every 9.987 milliseconds and yield a 100.1603 Hz sample rate.
91 #define CRVALUE 78
92 
93 
94 // Globals used to maintain ADC state and values.
95 volatile uint8_t adc_channel;
96 volatile uint8_t adc_power_ready;
97 volatile uint16_t adc_power_value;
98 volatile uint8_t adc_position_ready;
99 volatile uint16_t adc_position_value;
100 volatile uint8_t adc_voltage_needed;

103 void adc_init(void)
104 // Initialize ADC conversion for reading current monitoring and position.
105 {
106  // Read from position first.
107  adc_channel = ADC_CHANNEL_POSITION;
108 
109  // Initialize flags and values.
110  adc_power_ready = 0;
111  adc_power_value = 0;
112  adc_position_ready = 0;
113  adc_position_value = 0;
114  adc_voltage_needed = 1;
115 
116  //
117  // Initialize ADC registers to yield a 125KHz clock.
118  //
119 
120 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
121  // Make sure port PB4 (ADC3) and PB5 (ADC0) are set as input.
122  PORTB &= ~((1<<PB4) | (1<<PB5));
123 
124  // Disable digital input for ADC3 and ADC0 to reduce power consumption.
125  DIDR0 |= (1<<ADC3D) | (1<<ADC0D);
126 
127  // Set the ADC multiplexer selection register.
128  ADMUX = (0<<REFS2) | (0<<REFS1) | (0<<REFS0) | // Select VCC as voltage reference.
129  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (1<<MUX0) | // Select ADC3 (PB3), no gain.
130  (0<<ADLAR); // Keep high bits right adjusted.
131 
132  // Set the ADC control and status register B.
133  ADCSRB = (0<<BIN) | // Gain working in unipolar mode.
134  (0<<IPR) | // No input polarity reversal.
135  (0<<ADTS2) | (1<<ADTS1) | (1<<ADTS0); // Timer/Counter0 Compare Match A.
136 
137  // Set the ADC control and status register A.
138  ADCSRA = (1<<ADEN) | // Enable ADC.
139  (0<<ADSC) | // Don's start yet, will be auto triggered.
140  (1<<ADATE) | // Start auto triggering.
141  (1<<ADIE) | // Activate ADC conversion complete interrupt.
142  ADPS; // Prescale -- see above.
143 #endif // __AVR_ATtiny45__ || __AVR_ATtiny85____
144 
145 #if defined(__AVR_ATmega8__)
146  // Make sure ports PC0 (ADC0), PC1 (ADC1) and PC2 (ADC2) are set low.
147  PORTC &= ~((1<<PC2) | (1<<PC1) | (1<<PC0));
148 
149  // Set the ADC multiplexer selection register.
150  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
151  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (0<<MUX0) | // Select ADC2 (PC2) as analog input.
152  (0<<ADLAR); // Keep high bits right adjusted.
153 
154 
155  // Set the ADC control and status register A.
156  ADCSRA = (1<<ADEN) | // Enable ADC.
157  (1<<ADSC) | // Start the first conversion.
158  (0<<ADFR) | // Free running disabled.
159  (1<<ADIF) | // Clear any pending interrupt.
160  (1<<ADIE) | // Activate ADC conversion complete interrupt.
161  ADPS; // Prescale -- see above.
162 
163  // Reset the counter value to initiate another ADC sample at the specified time.
164  TCNT0 = 256 - CRVALUE;
165 #endif // __AVR_ATmega8____
166 
167 #if defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
168  // Make sure ports PC0 (ADC0), PC1 (ADC1) and PC2 (ADC2) are set low.
169  PORTC &= ~((1<<PC0) | (1<<PC1) | (1<<PC2));
170 
171  // Disable digital input for ADC0, ADC1 and ADC2 to reduce power consumption.
172  DIDR0 |= (1<<ADC2D) | (1<<ADC1D) |(1<<ADC0D);
173 
174  // Set the ADC multiplexer selection register.
175  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
176  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (0<<MUX0) | // Select ADC2 (PC2) as analog input.
177  (0<<ADLAR); // Keep high bits right adjusted.
178 
179  // Set the ADC control and status register B.
180  ADCSRB = (0<<ADTS2) | (1<<ADTS1) | (1<<ADTS0); // Timer/Counter0 Compare Match A.
181 
182  // Set the ADC control and status register A.
183  ADCSRA = (1<<ADEN) | // Enable ADC.
184  (0<<ADSC) | // Don's start yet, will be auto triggered.
185  (1<<ADATE) | // Start auto triggering.
186  (1<<ADIE) | // Activate ADC conversion complete interrupt.
187  ADPS; // Prescale -- see above.
188 #endif // __AVR_ATmega88__ || __AVR_ATmega168__
189 
190 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
191  // Set timer/counter0 control register A.
192  TCCR0A = (0<<COM0A1) | (0<<COM0A0) | // Disconnect OCOA.
193  (0<<COM0B1) | (0<<COM0B0) | // Disconnect OCOB.
194  (1<<WGM01) | (0<<WGM00); // Mode 2 - clear timer on compare match.
195 
196  // Set timer/counter0 control register B.
197  TCCR0B = (0<<FOC0A) | (0<<FOC0B) | // No force output compare A or B.
198  (0<<WGM02) | // Mode 2 - clear timer on compare match.
199  CSPS; // Timer clock prescale -- see above.
200 
201  // Set the timer/counter0 interrupt masks.
202  TIMSK = (1<<OCIE0A) | // Interrupt on compare match A.
203  (0<<OCIE0B) | // No interrupt on compare match B.
204  (0<<TOIE0); // No interrupt on overflow.
205 
206  // Set the compare match A value which initiates an ADC sample.
207  OCR0A = CRVALUE;
208 #endif // __AVR_ATtiny45__ || __AVR_ATtiny85__
209 
210 #if defined(__AVR_ATmega8__)
211  // Set timer/counter0 control register.
212  TCCR0 = CSPS; // Timer clock prescale -- see above.
213 
214  // Clear any pending interrupt.
215  TIFR |= (1<<TOV0); // Interrupt on overflow.
216 
217  // Set the timer/counter0 interrupt masks.
218  TIMSK |= (1<<TOIE0); // Interrupt on overflow.
219 #endif // __AVR_ATmega8____
220 
221 #if defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
222  // Set timer/counter0 control register A.
223  TCCR0A = (0<<COM0A1) | (0<<COM0A0) | // Disconnect OCOA.
224  (0<<COM0B1) | (0<<COM0B0) | // Disconnect OCOB.
225  (1<<WGM01) | (0<<WGM00); // Mode 2 - clear timer on compare match.
226 
227  // Set timer/counter0 control register B.
228  TCCR0B = (0<<FOC0A) | (0<<FOC0B) | // No force output compare A or B.
229  (0<<WGM02) | // Mode 2 - clear timer on compare match.
230  CSPS; // Timer clock prescale -- see above.
231 
232  // Set the timer/counter0 interrupt masks.
233  TIMSK0 = (1<<OCIE0A) | // Interrupt on compare match A.
234  (0<<OCIE0B) | // No interrupt on compare match B.
235  (0<<TOIE0); // No interrupt on overflow.
236 
237  // Set the compare match A value which initiates an ADC sample.
238  OCR0A = CRVALUE;
239 #endif // __AVR_ATmega88__ || __AVR_ATmega168__
240 }

243 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__) || defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)

245 SIGNAL(SIG_OUTPUT_COMPARE0A)
246 // Handles timer/counter0 compare match A.
247 {
248  // Increment the timer when positions are being sampled.
249  if (adc_channel == ADC_CHANNEL_POSITION) timer_increment();
250 }
251 
252 #endif // __AVR_ATtiny45__ || __AVR_ATtiny85__ || __AVR_ATmega88__ || __AVR_ATmega168__
253 
254 #if defined(__AVR_ATmega8__)
255 
256 SIGNAL(SIG_OVERFLOW0)
257 // Handles timer/counter0 overflow. This interrupts initiates the next
258 // ADC sample and assumes that the ADC sample will complete before the
259 // next timer overflow interrupt.
260 {
261  // Increment the timer when positions are being sampled.
262  if (adc_channel == ADC_CHANNEL_POSITION) timer_increment();
263 
264  // Initiate an ADC sample.
265  ADCSRA = (1<<ADEN) | // Enable ADC.
266  (1<<ADSC) | // Start the first conversion.
267  (0<<ADFR) | // Free running disabled.
268  (1<<ADIF) | // Clear any pending interrupt.
269  (1<<ADIE) | // Activate ADC conversion complete interrupt.
270  ADPS; // Prescale -- see above.
271 
272  // Reset the counter value to initiate another ADC sample at the specified time.
273  TCNT0 = 256 - CRVALUE;
274 }
275 
276 #endif // __AVR_ATmega8__
277 
278 SIGNAL(SIG_ADC)
279 // Handles ADC interrupt.
280 {
281  // Read the 10-bit ADC value.
282  uint16_t new_value = ADCW;
283 
284  // Which channel is being read?
285  switch (adc_channel)
286  {
287 
288  case ADC_CHANNEL_POSITION:
289 
290  // Save the new position value.
291  adc_position_value = new_value;
292 
293  // Flag the position value as ready.
294  adc_position_ready = 1;
295 
296  // Switch to power for the next reading.
297  adc_channel = ADC_CHANNEL_POWER;
298 
299 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
300  // Set the ADC multiplexer selection register.
301  ADMUX = (0<<REFS2) | (0<<REFS1) | (0<<REFS0) | // Select VCC as voltage reference.
302  (0<<MUX3) | (0<<MUX2) | (0<<MUX1) | (0<<MUX0) | // Select ADC0 (PB5), no gain.
303  (0<<ADLAR); // Keep high bits right adjusted.
304 
305  // Start the ADC of the power channel now
306  ADCSRA |= (1<<ADSC);
307 #endif
308 
309 #if defined(__AVR_ATmega8__) || defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
310  // Set the ADC multiplexer selection register.
311  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
312  (0<<MUX3) | (0<<MUX2) | (0<<MUX1) | (0<<MUX0) | // Select ADC0 (PC0) as analog input.
313  (0<<ADLAR); // Keep high bits right adjusted.
314 
315  // Start the ADC of the power channel now
316  ADCSRA |= (1<<ADSC);
317 #endif
318 
319  break;
320 
321 
322  case ADC_CHANNEL_POWER:
323 
324  // Save the new power value.
325  adc_power_value = new_value;
326 
327  // Flag the power value as ready.
328  adc_power_ready = 1;
329 
330 
331 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
332  // Switch to position for the next reading.
333  adc_channel = ADC_CHANNEL_POSITION;
334 
335 
336  // Set the ADC multiplexer selection register.
337  ADMUX = (0<<REFS2) | (0<<REFS1) | (0<<REFS0) | // Select VCC as voltage reference.
338  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (1<<MUX0) | // Select ADC3 (PB3), no gain.
339  (0<<ADLAR);
340 
341  break;
342 #endif
343 
344 #if defined(__AVR_ATmega8__) || defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
345 
346  if (adc_voltage_needed)
347  {
348  // Switch to voltage for the next reading.
349  adc_channel = ADC_CHANNEL_VOLTAGE;
350 
351  // Set the ADC multiplexer selection register.
352  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
353  (0<<MUX3) | (0<<MUX2) | (0<<MUX1) | (1<<MUX0) | // Select ADC1 (PC1) as analog input.
354  (0<<ADLAR); // Keep high bits right adjusted.
355 
356  // Start the ADC of the voltage channel now
357  ADCSRA |= (1<<ADSC);
358  }
359  else
360  {
361  // Switch to position for the next reading.
362  adc_channel = ADC_CHANNEL_POSITION;
363 
364  // Set the ADC multiplexer selection register.
365  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
366  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (0<<MUX0) | // Select ADC2 (PC2) as analog input.
367  (0<<ADLAR); // Keep high bits right adjusted.
368  }
369 
370  break;
371 
372 
373  case ADC_CHANNEL_VOLTAGE:
374 
375  // Remove flag
376  adc_voltage_needed = 0;
377 
378  // Save voltage value to registers
379  registers_write_word(REG_VOLTAGE_HI, REG_VOLTAGE_LO, new_value);
380 
381  // Switch to position for the next reading.
382  adc_channel = ADC_CHANNEL_POSITION;
383 
384  // Set the ADC multiplexer selection register.
385  ADMUX = (0<<REFS1) | (1<<REFS0) | // Select AVCC as voltage reference.
386  (0<<MUX3) | (0<<MUX2) | (1<<MUX1) | (0<<MUX0) | // Select ADC2 (PC2) as analog input.
387  (0<<ADLAR); // Keep high bits right adjusted.
388  break;
389 #endif
390 
391 
392  }
393 }
