112 static uint8_t twi_registers_read(uint8_t address)
113 // Read the byte from the specified register. This function handles the
114 // reading of special registers such as unused registers, redirect and
115 // redirected registers.
116 {
117  // Mask the most significant bit of the address.
118  address &= 0x7F;
119 
120  // Are we reading a normal register?
121  if (address <= MAX_WRITE_PROTECT_REGISTER)
122  {
123  // Yes. Complete the read.
124  return registers_read_byte(address);
125  }
126 
127  // Are we reading an unused register.
128  if (address <= MAX_UNUSED_REGISTER)
129  {
130  // Block the read.
131  return 0;
132  }
133 
134  // Are we reading a redirect register.
135  if (address <= MAX_REDIRECT_REGISTER)
136  {
137  // Yes. Complete the read.
138  return registers_read_byte(address - (MIN_REDIRECT_REGISTER - MIN_UNUSED_REGISTER));
139  }
140 
141  // Are we reading a redirected register?
142  if (address <= MAX_REDIRECTED_REGISTER)
143  {
144  // Adjust address to reference appropriate redirect register.
145  address = MIN_REDIRECT_REGISTER + (address - MIN_REDIRECTED_REGISTER);
146 
147  // Get the address from the redirect register.
148  address = registers_read_byte(address - (MIN_REDIRECT_REGISTER - MIN_UNUSED_REGISTER));
149 
150  // Prevent infinite recursion.
151  if (address <= MAX_REDIRECT_REGISTER)
152  {
153  // Recursively read redirected address.
154  return twi_registers_read(address);
155  }
156  }
157 
158  // All other reads are blocked.
159  return 0;
160 }

163 static void twi_registers_write(uint8_t address, uint8_t data)
164 // Write non-write protected registers. This function handles the
165 // writing of special registers such as unused registers, redirect and
166 // redirected registers.
167 {
168  // Mask the most significant bit of the address.
169  address &= 0x7F;
170 
171  // Are we writing a read only register?
172  if (address <= MAX_READ_ONLY_REGISTER)
173  {
174  // Yes. Block the write.
175  return;
176  }
177 
178  // Are we writing a read/write register?
179  if (address <= MAX_READ_WRITE_REGISTER)
180  {
181  // Yes. Complete the write.
182  registers_write_byte(address, data);
183 
184  return;
185  }
186 
187  // Is writing to the upper registers disabled?
188  if (registers_is_write_disabled())
189  {
190  // Yes. Block the write.
191  return;
192  }
193 
194  // Are we writing a write protected register?
195  if (address <= MAX_WRITE_PROTECT_REGISTER)
196  {
197  // Yes. Complete the write if writes are enabled.
198  registers_write_byte(address, data);
199 
200  return;
201  }
202 
203  // Are we writing an unused register.
204  if (address <= MAX_UNUSED_REGISTER)
205  {
206  // Yes. Block the write.
207  return;
208  }
209 
210 
211  // Are we writing a redirect register.
212  if (address <= MAX_REDIRECT_REGISTER)
213  {
214  // Yes. Complete the write.
215  registers_write_byte(address - (MIN_REDIRECT_REGISTER - MIN_UNUSED_REGISTER), data);
216 
217  return;
218  }
219 
220  // Are we writing a redirected register?
221  if (address <= MAX_REDIRECTED_REGISTER)
222  {
223  // Adjust address to reference appropriate redirect register.
224  address = MIN_REDIRECT_REGISTER + (address - MIN_REDIRECTED_REGISTER);
225 
226  // Get the address from the redirect register.
227  address = registers_read_byte(address - (MIN_REDIRECTED_REGISTER - MIN_UNUSED_REGISTER));
228 
229  // Prevent infinite recursion.
230  if (address <= MAX_REDIRECT_REGISTER)
231  {
232  // Recursively write redirected address.
233  twi_registers_write(address, data);
234 
235  return;
236  }
237  }
238 
239  // All other writes are blocked.
240  return;
241 }

245 static void twi_write_buffer(void)
246 // Write the recieve buffer to memory.
247 {
248  // Loop over the data within the write buffer.
249  for (twi_chk_count = 0; twi_chk_count < twi_chk_count_target; twi_chk_count++)
250  {
251  // Write the data to the addressed register.
252  twi_registers_write(twi_address, twi_chk_write_buffer[twi_chk_count & TWI_CHK_WRITE_BUFFER_MASK]);
253 
254  // Increment to the next address.
255  ++twi_address;
256  }
257 }

261 static uint8_t twi_read_data()
262 // Handle checked/non-checked read of data.
263 {
264  // By default read the data to be returned.
265  uint8_t data = twi_registers_read(twi_address);
266 
267 #if TWI_CHECKED_ENABLED
268  // Are we handling checked data?
269  if (twi_data_state == TWI_DATA_STATE_CHECKED_DATA)
270  {
271  // Have we reached the end of the read?
272  if (twi_chk_count < twi_chk_count_target)
273  {
274  // Add the data to the check sum.
275  twi_chk_sum += data;
276 
277  // Increment the check sum data count.
278  ++twi_chk_count;
279 
280  // Increment the address.
281  ++twi_address;
282  }
283  else
284  {
285  // Replace the data with the checksum.
286  data = twi_chk_sum;
287  }
288  }
289  else
290  {
291  // Increment the address.
292  ++twi_address;
293  }
294 #else
295  // Increment the address.
296  ++twi_address;
297 #endif
298 
299  return data;
300 }

303 static uint8_t twi_write_data(uint8_t data)
304 // Handle checked/non-checked write of data or command.
305 {
306  // By default, return ACK from write.
307  uint8_t ack = TWI_ACK;
308 
309  // Handle the write depending on the write state.
310  switch (twi_data_state)
311  {
312  case TWI_DATA_STATE_COMMAND:
313 
314  // This is a byte.
315  if (data < TWI_CMD_RESET)
316  {
317  // Capture the address.
318  twi_address = data;
319 
320  // Update the write state.
321  twi_data_state = TWI_DATA_STATE_DATA;
322  }
323 #if TWI_CHECKED_ENABLED
324  else if (data == TWI_CMD_CHECKED_TXN)
325  {
326  // Update the write state.
327  twi_data_state = TWI_DATA_STATE_CHECKED_COUNTING;
328  }
329 #endif
330  else
331  {
332  // Handle the command asynchronously.
333  twi_rxhead = (twi_rxhead + 1) & TWI_RX_BUFFER_MASK;
334  twi_rxbuf[twi_rxhead] = data;
335  }
336 
337  break;
338 
339  case TWI_DATA_STATE_DATA:
340 
341  // Write the data to the addressed register.
342  twi_registers_write(twi_address, data);
343 
344  // Increment to the next address.
345  ++twi_address;
346 
347  break;
348 
349 #if TWI_CHECKED_ENABLED
350  case TWI_DATA_STATE_CHECKED_COUNTING:
351 
352  // Read in the count (Make sure it's less than the max count)
353  // and start the checksum
354  twi_chk_sum = twi_chk_count_target = data & TWI_CHK_WRITE_BUFFER_MASK;
355 
356  // Clear the checksum and count.
357  twi_chk_count = 0;
358 
359  // Update the write state.
360  twi_data_state = TWI_DATA_STATE_CHECKED_ADDRESS;
361 
362  break;
363 
364  case TWI_DATA_STATE_CHECKED_ADDRESS:
365 
366  // Capture the address and include it in the checksum
367  twi_chk_sum += twi_address = data;
368 
369  // Update the write state.
370  twi_data_state = TWI_DATA_STATE_CHECKED_DATA;
371 
372  break;
373 
374  case TWI_DATA_STATE_CHECKED_DATA:
375 
376  // Have we reached the end of the write?
377  if (twi_chk_count < twi_chk_count_target)
378  {
379  // No. Write the data to the checksum buffer
380  twi_chk_write_buffer[twi_chk_count & TWI_CHK_WRITE_BUFFER_MASK] = data;
381 
382  // Add the data to the checksum.
383  twi_chk_sum += data;
384 
385  // Increment the check sum data count.
386  ++twi_chk_count;
387  }
388  else
389  {
390  // Verify the checksum
391  if (data == twi_chk_sum)
392  {
393  // Write the checksum buffer to addressed registers.
394  twi_write_buffer();
395  }
396  else
397  {
398  // Checksum failed so return NACK.
399  ack = TWI_NAK;
400  }
401  }
402 
403  break;
404 #endif
405  }
406 
407  return ack;
408 }

411 void
412 twi_slave_init(uint8_t slave_address)
413 // Initialise USI for TWI slave mode.
414 {
415  // Flush the buffers.
416  twi_rxtail = 0;
417  twi_rxhead = 0;
418 
419 #if defined(__AVR_ATtiny45__) || defined(__AVR_ATtiny85__)
420  // Set the slave address.
421  twi_slave_address = slave_address & 0x7f;
422 
423  // Set the interrupt enable, wire mode and clock settings. Note: At this
424  // time the wire mode must not be set to hold the SCL line low when the
425  // counter overflows. Otherwise, this TWI slave will interfere with other
426  // TWI slaves.
427  USICR = (0<<USISIE) | (0<<USIOIE) | // Disable start condition and overflow interrupt.
428  (1<<USIWM1) | (0<<USIWM0) | // Set USI to two-wire mode without clock stretching.
429  (1<<USICS1) | (0<<USICS0) | (0<<USICLK) | // Shift Register Clock Source = External, positive edge
430  (0<<USITC); // No toggle of clock pin.
431 
432  // Clear the interrupt flags and reset the counter.
433  USISR = (1<<USISIF) | (1<<USIOIF) | (1<<USIPF) | // Clear interrupt flags.
434  (0x0<<USICNT0); // USI to sample 8 bits or 16 edge toggles.
435 
436  // Configure SDA.
437  DDR_USI &= ~(1<<DD_SDA);
438  PORT_USI &= ~(1<<P_SDA);
439 
440  // Configure SCL.
441  DDR_USI |= (1<<DD_SCL);
442  PORT_USI |= (1<<P_SCL);
443 
444  // Start condition interrupt enable.
445  USICR |= (1<<USISIE);
446 #endif // __AVR_ATtiny45__ || __AVR_ATtiny85____
447 
448 #if defined(__AVR_ATmega8__) || defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
449  // Set own TWI slave address.
450  TWAR = slave_address << 1;
451 
452  // Default content = SDA released.
453  TWDR = 0xFF;
454 
455  // Initialize the TWI interrupt to wait for a new event.
456  TWCR = (1<<TWEN) | // Keep the TWI interface enabled.
457  (1<<TWIE) | // Keep the TWI interrupt enabled.
458  (0<<TWSTA) | // Don't generate start condition.
459  (0<<TWSTO) | // Don't generate stop condition.
460  (1<<TWINT) | // Clear the TWI interrupt.
461  (1<<TWEA) | // Acknowledge the data.
462  (0<<TWWC); //
463 #endif // __AVR_ATmega8__ || __AVR_ATmega88__ || __AVR_ATmega168__
464 }

468 // Returns a byte from the receive buffer. Waits if buffer is empty.

475 
476  // Return data from the buffer.
477  return twi_rxbuf[twi_rxtail];
478 }

491 SIGNAL(SIG_USI_START)
492 // Handle the TWI start condition. This is called when the TWI master initiates
493 // communication with a TWI slave by asserting the TWI start condition.
494 {
495  // Wait until the "Start Condition" is complete when SCL goes low. If we fail to wait
496  // for SCL to go low we may miscount the number of clocks pulses for the data because
497  // the transition of SCL could be mistaken as one of the data clock pulses.
498  while ((PIN_USI & (1<<PIN_SCL)));
499 
500  // Reset the overflow state.
501  twi_overflow_state = TWI_OVERFLOW_STATE_NONE;
502 
503  // Clear the interrupt flags and reset the counter.
504  USISR = (1<<USISIF) | (1<<USIOIF) | (1<<USIPF) | // Clear interrupt flags.
505  (0x00<<USICNT0); // USI to sample 8 bits or 16 edge toggles.
506 
507  // Update the interrupt enable, wire mode and clock settings.
508  USICR = (1<<USISIE) | (1<<USIOIE) | // Enable Overflow and Start Condition Interrupt.
509  (1<<USIWM1) | (1<<USIWM0) | // Maintain USI in two-wire mode with clock stretching.
510  (1<<USICS1) | (0<<USICS0) | (0<<USICLK) | // Shift Register Clock Source = External, positive edge
511  (0<<USITC); // No toggle of clock pin.
512 }

515 SIGNAL(SIG_USI_OVERFLOW)
516 // Handle the TWI overflow condition. This is called when the TWI 4-bit counter
517 // overflows indicating the TWI master has clocked in/out a databyte or a single
518 // ack/nack byte following a databyte transfer.
519 {
520  // Buffer the USI data.
521  uint8_t usi_data = USIDR;
522 
523  // Handle the interrupt based on the overflow state.
524  switch (twi_overflow_state)
525  {
526  // Handle the first byte transmitted from master -- the slave address.
527  case TWI_OVERFLOW_STATE_NONE:
528 
529  // Are we receiving our address?
530  if ((usi_data >> 1) == twi_slave_address)
531  {
532  // Are we transmitting or receiving data?
533  if (usi_data & 0x01)
534  {
535  // We are to transmit data. Reset the overflow
536  // state, but preserve the data state from the last write.
537  twi_overflow_state = TWI_OVERFLOW_STATE_ACK_PR_TX;
538  }
539  else
540  {
541  // We are receiving data. Set data and overflow state.
542  twi_data_state = TWI_DATA_STATE_COMMAND;
543  twi_overflow_state = TWI_OVERFLOW_STATE_ACK_PR_RX;
544  }
545 
546  // Set SDA for output.
547  PORT_USI |= (1<<P_SDA);
548  DDR_USI |= (1<<DD_SDA);
549 
550  // Load data for ACK.
551  USIDR = TWI_ACK;
552 
553  // Reload counter for ACK -- two clock transitions.
554  USISR = 0x0E;
555  }
556  else
557  {
558  // No. Reset USI to detect start condition. Update the interrupt enable,
559  // wire mode and clock settings. Note: At this time the wire mode must
560  // not be set to hold the SCL line low when the counter overflows.
561  // Otherwise, this TWI slave will interfere with other TWI slaves.
562  USICR = (1<<USISIE) | (0<<USIOIE) | // Enable Start Condition Interrupt. Disable overflow.
563  (1<<USIWM1) | (0<<USIWM0) | // Maintain USI in two-wire mode without clock stretching.
564  (1<<USICS1) | (0<<USICS0) | (0<<USICLK) | // Shift Register Clock Source = External, positive edge
565  (0<<USITC); // No toggle of clock pin.
566  }
567 
568  break;
569 
570  // Ack sent to master so prepare to receive more data.
571  case TWI_OVERFLOW_STATE_ACK_PR_RX:
572 
573  // Update our state.
574  twi_overflow_state = TWI_OVERFLOW_STATE_DATA_RX;
575 
576  // Set SDA for input
577  DDR_USI &= ~(1<<DD_SDA);
578  PORT_USI &= ~(1<<P_SDA);
579 
580  break;
581 
582  // Data received from master so prepare to send ACK.
583  case TWI_OVERFLOW_STATE_DATA_RX:
584 
585  // Update our state.
586  twi_overflow_state = TWI_OVERFLOW_STATE_ACK_PR_RX;
587 
588  // Write the data and return ack/nack.
589  USIDR = twi_write_data(usi_data);
590 
591  // Set SDA for output.
592  PORT_USI |= (1<<P_SDA);
593  DDR_USI |= (1<<DD_SDA);
594 
595  // Reload counter for ACK -- two clock transitions.
596  USISR = 0x0E;
597 
598  break;
599 
600  // ACK received from master. Reset USI state if NACK received.
601  case TWI_OVERFLOW_STATE_PR_ACK_TX:
602 
603  // Check the lowest bit for NACK? If set, the master does not want more data.
604  if (usi_data & 0x01)
605  {
606  // Reset USI to detect start condition. Update the interrupt enable,
607  // wire mode and clock settings. Note: At this time the wire mode must
608  // not be set to hold the SCL line low when the counter overflows.
609  // Otherwise, this TWI slave will interfere with other TWI slaves.
610  USICR = (1<<USISIE) | (0<<USIOIE) | // Enable Start Condition Interrupt. Disable overflow.
611  (1<<USIWM1) | (0<<USIWM0) | // Maintain USI in two-wire mode without clock stretching.
612  (1<<USICS1) | (0<<USICS0) | (0<<USICLK) | // Shift Register Clock Source = External, positive edge
613  (0<<USITC); // No toggle of clock pin.
614 
615 
616  // Clear the overflow interrupt flag and release the hold on SCL.
617  USISR |= (1<<USIOIF);
618 
619  return;
620  }
621 
622  // Handle sending a byte of data.
623  case TWI_OVERFLOW_STATE_ACK_PR_TX:
624 
625  // Update our state.
626  twi_overflow_state = TWI_OVERFLOW_STATE_DATA_TX;
627 
628  // Set SDA for output.
629  PORT_USI |= (1<<P_SDA);
630  DDR_USI |= (1<<DD_SDA);
631 
632  // Read the checked/non-checked data.
633  USIDR = twi_read_data();
634 
635  break;
636 
637  // Data sent to to master so prepare to receive ack.
638  case TWI_OVERFLOW_STATE_DATA_TX:
639 
640  // Update our state.
641  twi_overflow_state = TWI_OVERFLOW_STATE_PR_ACK_TX;
642 
643  // Set SDA for input.
644  DDR_USI &= ~(1<<DD_SDA);
645  PORT_USI &= ~(1<<P_SDA);
646 
647  // Reload counter for ACK -- two clock transitions.
648  USISR = 0x0E;
649 
650  break;
651 
652  }
653 
654  // Clear the overflow interrupt flag and release the hold on SCL.
655  USISR |= (1<<USIOIF);
656 }

667 SIGNAL(SIG_TWI)
668 // Handle the TWI interrupt condition.
669 {
670  switch (TWSR)
671  {
672  // Own SLA+R has been received; ACK has been returned.
673  case TWI_STX_ADR_ACK:
674  // Data byte in TWDR has been transmitted; ACK has been received.
675  case TWI_STX_DATA_ACK:
676

678  TWDR = twi_read_data();
679 
680  // Data byte will be transmitted and ACK should be received.
681  TWCR = (1<<TWEN) | // Keep the TWI interface enabled.

693  case TWI_STX_DATA_ACK_LAST_BYTE:

727

729  TWCR = (1<<TWEN) | // Keep the TWI interface enabled.

740  case TWI_SRX_ADR_DATA_NACK:
741  // A STOP condition or repeated START condition has been received while still addressed as Slave.
742  case TWI_SRX_STOP_RESTART:
743

754 
755  // Bus error due to an illegal START or STOP condition.
756  case TWI_BUS_ERROR:
757

759  // In all cases, the bus is released and TWSTO is cleared.

768

770  case TWI_NO_STATE:
771 
772  // No action required.
773  break;
774 
775  default:
776 
777  // Reset the TWI interrupt to wait for a new event.
778  TWCR = (1<<TWEN) | // Keep the TWI interface enabled.
