37 // The minimum and maximum servo position as defined by 10-bit ADC values.

48 // The accumulator is a structure arranged so that both the most significant
49 // and least significant words can be indepently accessed. The accumulator
50 // maintains 32-bit resolution, but only 16-bits are needed for output.
51 static union
52 {
53  struct
54  {
55  int16_t lo;
56  int16_t hi;
57  } small;
58  int32_t big;
59 } integral_accumulator;

62 static int16_t fixed_multiply(int16_t component, uint16_t fixed_gain)
63 // Multiplies the signed 16-bit component by the unsigned 16-bit fixed
64 // point gain to return a 32-bit output. The result is scaled to account
65 // for the fixed point value.
66 {
67  int32_t output;
68 
69  // Multiply the value by the fixed gain value.
70  output = (int32_t) component * (int32_t) fixed_gain;
71 
72  // Scale result by 256 to account for fixed point gain.
73  output /= 256;
74 
75  return (int16_t) output;
76 }

79 static inline int16_t integral_accumulator_get(void)
80 // This function returns the most significant word of the integral gain.
81 {

86 static inline void integral_accumulator_update(int16_t command_error, uint16_t integral_gain)
87 // This function updates the integral accumulator with the command error.
88 // Note: The command error is maintained with 32 bit precision, although
89 // only the upper 16 bits are used for output calculations.
90 {
91  int32_t temp;
92 
93  // Multiply the command error by the fixed integral gain value.
94  temp = (int32_t) command_error * (int32_t) integral_gain;
95 
96  // Add to the accumulator adjusting for multiplication.
97  integral_accumulator.big += temp;
98 }

101 static inline void integral_accumulator_reset(int16_t new_value)
102 // This function resets the integral accumulator to a new value.
103 {
104  integral_accumulator.small.hi = new_value;
105  integral_accumulator.small.lo = 0;

135 int16_t ipd_position_to_pwm(int16_t current_position)
136 // This function takes the current servo position as input and outputs a pwm
137 // value for the servo motors. The current position value must be within the
138 // range 0 and 1023. The output will be within the range of -255 and 255 with
139 // values less than zero indicating clockwise rotation and values more than
140 // zero indicating counter-clockwise rotation.
141 //
142 // The feedback approach implemented here was first published in Richard Phelan's
143 // Automatic Control Systems, Cornell University Press, 1977 (ISBN 0-8014-1033-9)
144 //
145 // The theory of operation of this function will be filled in later, but the
146 // diagram below should gives a general picture of how it is intended to work.
147 //
148 //
149 // +<------- bounds checking -------+
150 // | |
151 // |¯¯¯¯¯| |¯¯¯¯¯¯¯¯| |¯¯¯¯¯| |¯¯¯¯¯¯¯¯¯| |
152 // command -->| - |-->|integral|-->| - |-->| motor |-->+-> actuator
153 // |_____| |________| |_____| |_________| |
154 // | | |
155 // | +<-- Kv * velocity --+
156 // | | |
157 // | +<-- Kp * position --+
158 // | |
159 // +<-------------Ki * position ---------------+
160 //
161 //
162 {
163  int16_t deadband;
164  int16_t command_position;
165  int16_t maximum_position;
166  int16_t minimum_position;
167  int16_t current_velocity;
168  int16_t command_error;
169  int16_t output;
170  int16_t position_output;
171  int16_t velocity_output;
172  int16_t integral_output;
173  uint16_t position_gain;
174  uint16_t velocity_gain;
175  uint16_t integral_gain;
176 
177  // Determine the velocity as the difference between the current and previous position.
178  current_velocity = current_position - previous_position;
179 
180  // Update the previous position.
181  previous_position = current_position;
182 
183  // Get the command position to where the servo is moving to from the registers.
184  command_position = (int16_t) registers_read_word(REG_SEEK_POSITION_HI, REG_SEEK_POSITION_LO);
185  minimum_position = (int16_t) registers_read_word(REG_MIN_SEEK_HI, REG_MIN_SEEK_LO);
186  maximum_position = (int16_t) registers_read_word(REG_MAX_SEEK_HI, REG_MAX_SEEK_LO);
187 
188  // Set the deadband value and divide by two for calculations below.
189  deadband = 2;
190 
191  // Are we reversing the seek sense?
192  if (registers_read_byte(REG_REVERSE_SEEK) != 0)
193  {
194  // Yes. Update the system registers with an adjusted reverse sense
195  // position. With reverse sense, the position value to grows from
196  // a low value to high value in the clockwise direction.
197  registers_write_word(REG_POSITION_HI, REG_POSITION_LO, (uint16_t) (MAX_POSITION - current_position));
198 
199  // Adjust command position for the reverse sense.
200  command_position = MAX_POSITION - command_position;
201  minimum_position = MAX_POSITION - minimum_position;
202  maximum_position = MAX_POSITION - maximum_position;
203  }
204  else
205  {
206  // No. Update the system registers with a non-reverse sense position.
207  // Normal position value grows from a low value to high value in the
208  // counter-clockwise direction.
209  registers_write_word(REG_POSITION_HI, REG_POSITION_LO, (uint16_t) current_position);
210  }
211 
212  // Sanity check the command position. We do this because this value is
213  // passed from the outside to the servo and it could be an invalid value.
214  if (command_position < minimum_position) command_position = minimum_position;
215  if (command_position > maximum_position) command_position = maximum_position;
216 
217  // The command error is the difference between the command position and current position.
218  command_error = command_position - current_position;
219 
220  // Adjust proportional error due to deadband. The potentiometer readings are a
221  // bit noisy and there is typically one or two units of difference from reading
222  // to reading when the servo is holding position. Adding deadband decreases some
223  // of the twitchiness in the servo caused by this noise.
224  if (command_error > deadband)
225  {
226  // Factor out deadband from the command error.
227  command_error -= deadband;
228  }
229  else if (command_error < -deadband)
230  {
231  // Factor out deadband from the command error.
232  command_error += deadband;
233  }
234  else
235  {
236  // Adjust to command error to zero within deadband.
237  command_error = 0;
238  }
239 
240  // Get the positional, velocity and integral gains from the registers.
241  position_gain = registers_read_word(REG_PID_PGAIN_HI, REG_PID_PGAIN_LO);
242  velocity_gain = registers_read_word(REG_PID_DGAIN_HI, REG_PID_DGAIN_LO);
243  integral_gain = registers_read_word(REG_PID_IGAIN_HI, REG_PID_IGAIN_LO);
244 
245  // Add the command error scaled by the position gain to the integral accumulator.
246  // The integral accumulator maintains a sum of total error over each iteration.
247  integral_accumulator_update(command_error, integral_gain);
248 
249  // Get the integral output. There is no gain applied to the integral output.
250  integral_output = integral_accumulator_get();
251 
252  // Determine the position output component. We multiply the position gain
253  // by the current position of the servo to create the position output.
254  position_output = fixed_multiply(current_position, position_gain);
255 
256  // Determine the velocity output component. We multiply the velocity gain
257  // by the current velocity of the servo to create the velocity output.
258  velocity_output = fixed_multiply(current_velocity, velocity_gain);
259 
260  // registers_write_word(REG_RESERVED_18, REG_RESERVED_19, (uint16_t) position_output);
261  // registers_write_word(REG_RESERVED_1A, REG_RESERVED_1B, (uint16_t) velocity_output);
262  // registers_write_word(REG_RESERVED_1C, REG_RESERVED_1D, (uint16_t) integral_output);
263 
264  // The integral output drives the output and the position and velocity outputs
265  // function as a frictional component to counter the integral output.
266  output = integral_output - position_output - velocity_output;
267 
268  // Is the output saturated? If so we need limit the output and clip
269  // the integral accumulator just at the saturation level.
270  if (output < MIN_OUTPUT)
271  {
272  // Calculate a new integral accumulator based on the output range. This value
273  // is calculated to keep the integral output just on the verge of saturation.
274  integral_accumulator_reset(position_output + velocity_output + MIN_OUTPUT);
275 
276  // Limit the output.
277  output = MIN_OUTPUT;
278  }
279  else if (output > MAX_OUTPUT)
280  {
281  // Calculate a new integral accumulator based on the output range. This value
282  // is calculated to keep the integral output just on the verge of saturation.
283  integral_accumulator_reset(position_output + velocity_output + MAX_OUTPUT);
284 
285  // Limit the output.
286  output = MAX_OUTPUT;
287  }
288 
289  // Return the output.
290  return output;
291 }
