38 // This file is intended to consolidate configuration settings
39 // for the OpenServo software into a single location. Enabling
40 // certain combinations of settings may result in a servo that
41 // doesn't function properly or be too large to fit within the
42 // 3K Flash limit for applications on the ATtiny45 MCU in the
43 // OpenServo. Where such limitations are known, comments should
44 // the placed within this file.
45 
46 // Enable (1) or disable (0) checksum verification within the
47 // twi.c module. When enabled the TWI_CMD_CHECKED_TXN command
48 // is enabled and basic checksum validation of reads and writes
49 // of registers can be made for more robust communication with
50 // the OpenServo. The checksum code consumes about 280 bytes
51 // of Flash code space when enabled.
52 #define TWI_CHECKED_ENABLED 0
53 
54 // Enable (1) or disable (0) the PID algorithm for motion
55 // control in the motion.c module. This setting cannot be
56 // set when the other XXX_MOTION_ENABLED flags are set.
57 //
58 // NOTE: This is the motion control algorithm most people should
59 // use until the other algorithms are further developed.
60 #define PID_MOTION_ENABLED 1
61 
62 // Enable (1) or disable (0) the IPD algorithm for motion
63 // control in the motion.c module. This setting cannot be
64 // set when the other XXX_MOTION_ENABLED flags are set.
65 //
66 // NOTE: The IPD algorithm is still under development and is
67 // currently unstable. Please contact Mike Thompson in the
68 // OpenServo forums for more information before enabling this
69 // feature.
70 #define IPD_MOTION_ENABLED 0
71 
72 // Enable (1) or disable (0) the state regulator algorithm
73 // for motion control in the motion.c module. This setting
74 // cannot be set when the other XXX_MOTION_ENABLED flags are
75 // set.
76 //
77 // NOTE: The state regulator code is still under development.
78 // Please contact Stefan Engelke in the OpenServo forums for
79 // more information before enabling this feature.
80 #define REGULATOR_MOTION_ENABLED 0
81 
82 // Enable (1) or disable (0) the Luenberg state estimator
83 // algorithm for determining servo speed. It is a realtime
84 // simulation of the servo behavior with a internal controller
85 // which keeps track of the difference between the simulated
86 // position and the measured one. This must be enabled when
87 // the MOTION_REGULATOR_ENABLED flag is enabled.
88 #define ESTIMATOR_ENABLED (REGULATOR_MOTION_ENABLED)
89 
90 // Enable (1) or disable (0) fixed point utility functions in
91 // the math.c module. Currently these routines are only used
92 // by the state estimator and state regulator.
93 #define FIXED_MATH_ENABLED (ESTIMATOR_ENABLED || REGULATOR_ENABLED)
94 
95 // Enable (1) or disable (0) cubic Hermite curve following motion
96 // functions. These functions allow the OpenServo to be controlled
97 // by keypoints along a cubic Hermite curve with each keypoint
98 // indicating servo position and velocity at a particular time.
99 #define CURVE_MOTION_ENABLED 1
100 
101 // Enable (1) or disable (0) some test motion code within the
102 // main.c module. This test code can be enabled to test basic
103 // positioning of the OpenServo without a TWI master controlling
104 // the OpenServo. It should normally be disabled.
105 #define MAIN_MOTION_TEST_ENABLED 0
106 
107 // Enable (1) or disable (0) standard servo pulse control signaling
108 // of the seek position. As of 2/2/07 this feature is still under
109 // active development. Please visit the OpenServo forums for the
110 // current status of this feature.
111 #define PULSE_CONTROL_ENABLED 0
112 
113 // Enable (1) or disable (0) the swapping of PWM output A and B.
114 // This swapping must sometimes enabled depending on whether the
115 // positive lead to the motor is attached to MOSFET/PWM output A
116 // or MOSFET/PWM output B. This option makes this easy to control
117 // within software.
118 #define SWAP_PWM_DIRECTION_ENABLED 0
119 
120 // Perform some sanity check of settings here.
121 #if PID_MOTION_ENABLED && (IPD_MOTION_ENABLED || REGULATOR_MOTION_ENABLED)
122 # error "Conflicting configuration settings for PID_MOTION_ENABLED"
123 #endif
124 #if IPD_MOTION_ENABLED && (PID_MOTION_ENABLED || REGULATOR_MOTION_ENABLED)
125 # error "Conflicting configuration settings for MOTION_IPD_ENABLED"
126 #endif
127 #if REGULATOR_MOTION_ENABLED && (PID_MOTION_ENABLED || IPD_MOTION_ENABLED)
128 # error "Conflicting configuration settings for REGULATOR_MOTION_ENABLED"
129 #endif
130 #if REGULATOR_MOTION_ENABLED && !ESTIMATOR_ENABLED
131 # error "Configuration settings for REGULATOR_MOTION_ENABLED requires ESTIMATOR_ENABLED."
132 #endif
133 #if CURVE_MOTION_ENABLED && PULSE_CONTROL_ENABLED
134 # warning "Conflicting configuration settings for CURVE_MOTION_ENABLED and PULSE_CONTROL_ENABLED"
135 #endif
