30 static inline int16_t shift_right16(int16_t val, uint8_t cnt)
31 // Arithmetic right shift of the signed 16-bit value.
32 {
33  __asm__ volatile (
34  "L_asr1%=:" "\n\t"
35  "cp %1,__zero_reg__" "\n\t"
36  "breq L_asr2%=" "\n\t"
37  "dec %1" "\n\t"
38  "asr %B0" "\n\t"
39  "ror %A0" "\n\t"
40  "rjmp L_asr1%=" "\n\t"
41  "L_asr2%=:" "\n\t"
42  : "=&r" (val)
43  : "r" (cnt), "0" (val)
44  );
45 
46  return val;
47 }

49 static inline int32_t shift_right32(int32_t val, uint8_t cnt)
50 // Arithmetic right shift of the signed 32-bit value.
51 {
52  __asm__ volatile (
53  "L_asr1%=:" "\n\t"
54  "cp %1,__zero_reg__" "\n\t"
55  "breq L_asr2%=" "\n\t"
56  "dec %1" "\n\t"
57  "asr %D0" "\n\t"
58  "ror %C0" "\n\t"
59  "ror %B0" "\n\t"
60  "ror %A0" "\n\t"
61  "rjmp L_asr1%=" "\n\t"
62  "L_asr2%=:" "\n\t"
63  : "=&r" (val)
64  : "r" (cnt), "0" (val)
65  );
66 
67  return val;
68 }

72 // Multiplies the two 16-bit values num1 and num2 into a 32-bit
73 // product. The product is then right shifted by the specified
74 // amount and the lower 16-bits of the value returned. The purpose
75 // of this function is to facilitate fixed point arithmetic. When
76 // multiplying two fixed point numbers with fixed points at bit
77 // fp_1 and fp_2 (counted from the least significant bit), the
78 // result will have the fixed point bit at:
79 //
80 // fp_result = fp_1 + fp_2
81 //
82 // To fit to a desired fixed point precision fp_3 the result must
83 // be shifted to the right by:
84 //
85 // shift = fp_result - fp_3
86 //
