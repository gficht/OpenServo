32 static inline void timer_set(uint16_t value)
33 {
34  // Set the timer to the specified value.

36 }

38 static inline uint16_t timer_get(void)
39 {
40  // Get the timer value.
41  return registers_read_word(REG_TIMER_HI, REG_TIMER_LO);
42 }
43 
44 static inline uint16_t timer_delta(uint16_t time_stamp)
45 {
46  uint16_t delta_time;
47 
48  // Get the current time.
49  uint16_t current_time = registers_read_word(REG_TIMER_HI, REG_TIMER_LO);
50 
51  // Determine the time delta from the time stamp.
52  if (current_time > time_stamp)
53  delta_time = current_time - time_stamp;
54  else
55  delta_time = 0xffff - (time_stamp - current_time) + 1;
56 
57  return delta_time;
58 }
59 
60 static inline void timer_increment(void)
61 {
62  uint16_t value;
63 
64  // Read the timer value.
65  value = registers_read_word(REG_TIMER_HI, REG_TIMER_LO);
66 
67  // Increment the timer value.
68  ++value;
69 
70  // Write the increment timer value.
71  registers_write_word(REG_TIMER_HI, REG_TIMER_LO, value);
72 }
