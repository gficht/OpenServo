87 // Initialize the curve buffer.
88 {
89  // Initialize the counter.
90  motion_counter = 0;
91 
92  // Initialize the duration.
93  motion_duration = 0;
94 
95  // Initialize the queue.
96  motion_head = 0;
97  motion_tail = 0;
98 
99  // Initialize the keypoint.
100  keys[0].delta = 0;
101  keys[0].position = 512.0;
102  keys[0].in_velocity = 0.0;
103  keys[0].out_velocity = 0.0;
104 
105  // Initialize an empty hermite curve at the center servo position.
106  curve_init(0, 0, 512.0, 512.0, 0.0, 0.0);
107 
108  // Reset the registers.
109  motion_registers_reset();

141 void motion_registers_reset(void)
142 // Reset the motion registers to zero values.
143 {
144  // Set the default position, velocity and delta data.
145  registers_write_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO, 0);
146  registers_write_word(REG_CURVE_IN_VELOCITY_HI, REG_CURVE_IN_VELOCITY_LO, 0);
147  registers_write_word(REG_CURVE_OUT_VELOCITY_HI, REG_CURVE_OUT_VELOCITY_LO, 0);
148  registers_write_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO, 0);
149 
150  // Update the buffer status.
151  registers_write_byte(REG_CURVE_RESERVED, 0);
152  registers_write_byte(REG_CURVE_BUFFER, motion_buffer_left());
153 }

156 uint8_t motion_append(void)
157 // Append a new curve keypoint from data stored in the curve registers. The keypoint
158 // is offset from the previous curve by the specified delta. An error is returned if
159 // there is no more room to store the new keypoint in the buffer or if the delta is
160 // less than one (a zero delta is not allowed).
161 {
162  int16_t position;
163  int16_t in_velocity;
164  int16_t out_velocity;
165  uint8_t next;
166  uint16_t delta;
167 
168  // Get the next index in the buffer.
169  next = (motion_head + 1) & MOTION_BUFFER_MASK;
170 
171  // Return error if we have looped the head to the tail and the buffer is filled.
172  if (next == motion_tail) return 0;
173 
174  // Get the position, velocity and time delta values from the registers.
175  position = (int16_t) registers_read_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO);
176  in_velocity = (int16_t) registers_read_word(REG_CURVE_IN_VELOCITY_HI, REG_CURVE_IN_VELOCITY_LO);
177  out_velocity = (int16_t) registers_read_word(REG_CURVE_OUT_VELOCITY_HI, REG_CURVE_OUT_VELOCITY_LO);
178  delta = (uint16_t) registers_read_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO);
179 
180  // Keypoint delta must be greater than zero.
181  if (delta < 1) return 0;
182 
183  // Fill in the next keypoint.
184  keys[next].delta = delta;
185  keys[next].position = int_to_float(position);
186  keys[next].in_velocity = fixed_to_float(in_velocity);
187  keys[next].out_velocity = fixed_to_float(out_velocity);
188 
189  // Is this keypoint being added to an empty buffer?
190  if (motion_tail == motion_head)
191  {
192  // Initialize a new hermite curve that gets us from the current position to the new position.
193  // We use a velocity of zero at each end to smoothly transition from one to the other.
194  curve_init(0, delta, curve_get_p1(), keys[next].position, 0.0, 0.0);
195  }
196 
197  // Increase the duration of the buffer.
198  motion_duration += delta;
199 
200  // Set the new head index.
201  motion_head = next;
202 
203  // Reset the motion registers and update the buffer status.
204  motion_registers_reset();
205 
206  return 1;
207 }

210 void motion_next(uint16_t delta)
211 // Increment the buffer counter by the indicated delta and return the position
212 // and velocity from the buffered curves. If the delta is zero the current
213 // position and velocity is returned.
214 {
215  float fposition;
216  float fvelocity;
217 
218  // Determine if curve motion is disabled in the registers.
219  if (!(registers_read_byte(REG_FLAGS_LO) & (1<<FLAGS_LO_MOTION_ENABLED))) return;
220 
221  // Are we processing an empty curve?
222  if (motion_tail == motion_head)
223  {
224  // Yes. Keep the counter and duration at zero.
225  motion_counter = 0;
226  motion_duration = 0;
227  }
228  else
229  {
230  // Increment the counter.
231  motion_counter += delta;
232 
233  // Have we exceeded the duration of the currently buffered curve?
234  while (motion_counter > curve_get_duration())
235  {
236  // Reduce the buffer counter by the currently buffered curve duration.
237  motion_counter -= curve_get_duration();
238 
239  // Reduce the buffer duration by the currently buffered curve duration.
240  motion_duration -= curve_get_duration();
241 
242  // Increment the tail to process the next buffered curve.
243  motion_tail = (motion_tail + 1) & MOTION_BUFFER_MASK;
244 
245  // Has the tail caught up with the head?
246  if (motion_tail == motion_head)
247  {
248  // Initialize an empty hermite curve with a zero duration. This is a degenerate case for
249  // the hermite cuve that will always return the position of the curve without velocity.
250  curve_init(0, 0, keys[motion_head].position, keys[motion_head].position, 0.0, 0.0);
251 
252  // Reset the buffer counter and duration to zero.
253  motion_counter = 0;
254  motion_duration = 0;
255  }
256  else
257  {
258  uint8_t curr_point;
259  uint8_t next_point;
260 
261  // Get the current point and next point for the curve.
262  curr_point = motion_tail;
263  next_point = (curr_point + 1) & MOTION_BUFFER_MASK;
264 
265  // Initialize the hermite curve from the current and next point.
266  curve_init(0, keys[next_point].delta,
267  keys[curr_point].position, keys[next_point].position,
268  keys[curr_point].out_velocity, keys[next_point].in_velocity);
269  }
270 
271  // Update the space available in the buffer.
272  registers_write_byte(REG_CURVE_BUFFER, motion_buffer_left());
273  }
274  }
275 
276  // Get the position and velocity from the hermite curve.
277  curve_solve(motion_counter, &fposition, &fvelocity);
278 
279  // The velocity is in position units a millisecond, but we really need the
280  // velocity to be measured in position units every 10 milliseconds to match
281  // the sample period of the ADC.
282  fvelocity *= 10.0;
283 
284  // Update the seek position register.
285  registers_write_word(REG_SEEK_POSITION_HI, REG_SEEK_POSITION_LO, float_to_int(fposition));
286 
287  // Update the seek velocity register.
288  registers_write_word(REG_SEEK_VELOCITY_HI, REG_SEEK_VELOCITY_LO, float_to_int(fvelocity));
289 }

292 uint8_t motion_buffer_left(void)
293 // The motion buffer can contain up to MOTION_BUFFER_SIZE keypoints. The function
294 // returns how many keypoints remain free in the buffer for use.
295 {
296  uint8_t space_left;
297 
298  // Determine the points left to store curve data.
299  if (motion_head < motion_tail)
300  {
301  space_left = (MOTION_BUFFER_SIZE - 1) - (MOTION_BUFFER_SIZE + motion_head - motion_tail);
302  }
303  else
304  {
305  space_left = (MOTION_BUFFER_SIZE - 1) - (motion_head - motion_tail);
306  }
307 
308  return space_left;
309 }
