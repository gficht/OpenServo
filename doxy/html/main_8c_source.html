61 static void config_pin_defaults(void)
62 // Configure pins to their default states to conform to recommendation that all
63 // AVR MCU pins have a defined level. We do this by configuring unused pins
64 // as inputs and enabling the internal pull-ups.
65 {
66 #if defined(__AVR_ATmega8__) || defined(__AVR_ATmega88__) || defined(__AVR_ATmega168__)
67  // Configure unused port B pins as inputs and enable internal pull-up resistor.
69  DDRB = (0<<DDB7) | (0<<DDB6) | (0<<DDB5) | (1<<DDB4) |
70  (1<<DDB3) | (1<<DDB2) | (1<<DDB1) | (0<<DDB0);
71  PORTB = (1<<PORTB7) | (1<<PORTB6) | (1<<PORTB5) |
72  (1<<PORTB4) | (1<<PORTB3) | //rs485 go to receive
73  (0<<PORTB2) | (0<<PORTB1) | (1<<PORTB0);
81  // Configure unused port C pins as inputs and enable internal pull-up resistor.
82  DDRC = (0<<DDC6) | (0<<DDC5) | (0<<DDC4) |
83  (0<<DDC3) | (0<<DDC2) | (0<<DDC1) | (0<<DDC0);
84  PORTC = (1<<PORTC6) | (1<<PORTC5) | (1<<PORTC4) |
85  (1<<PORTC3) | (1<<PORTC2) | (1<<PORTC1) | (1<<PORTC0);
86 
87 // Configure port D pins as inputs and enable internal pull-up resistor.
89  DDRD = (0<<DDD7) | (0<<DDD6) | (0<<DDD5) | (0<<DDD4) |
90  (0<<DDD3) | (0<<DDD2);// | (0<<DDD1) | (0<<DDD0);
91  PORTD = (1<<PORTD7) | (1<<PORTD6) | (1<<PORTD5) | (1<<PORTD4) |
92  (1<<PORTD3) | (1<<PORTD2);
101 #endif
102 }

107  // Configure pins to the default states.
108  config_pin_defaults();
109 
110  // Initialize the watchdog module.
111  watchdog_init();
112 
113  // First, initialize registers that control servo operation.
114  registers_init();
115 
116  // Initialize the PWM module.
117  pwm_init();
118 
119  // Initialize the ADC module.
120  adc_init();
121 
122 #if ESTIMATOR_ENABLED
123  // Initialize the state estimator module.
124  estimator_init();
125 #endif
126 
127 #if REGULATOR_MOTION_ENABLED
128  // Initialize the regulator algorithm module.
129  regulator_init();
130 #endif
131 
132 #if PID_MOTION_ENABLED
133  // Initialize the PID algorithm module.
134  pid_init();
135 #endif
136 
137 #if IPD_MOTION_ENABLED
138  // Initialize the IPD algorithm module.
139  ipd_init();
140 #endif
141 
142 #if CURVE_MOTION_ENABLED
143  // Initialize curve motion module.
144  motion_init();
145 #endif
146 
147  // Initialize the power module.
148  power_init();
149 
150 #if PULSE_CONTROL_ENABLED
151  pulse_control_init();
152 #endif
153 
154  UartInitRs485(&PORTD,PD2);
156  UartInit();
157 
158 
159  // Initialize the TWI slave module.
160  //twi_slave_init(registers_read_byte(REG_TWI_ADDRESS));
161 
162  // Finally initialize the timer.
163  timer_set(0);
164 
165  // Enable interrupts.
166  sei();
167 
168  // Wait until initial position value is ready.
169  while (!adc_position_value_is_ready());
170 
171 #if CURVE_MOTION_ENABLED
172  // Reset the curve motion with the current position of the servo.
173  motion_reset(adc_get_position_value());
174 #endif
175 
176  // Set the initial seek position and velocity.
177  registers_write_word(REG_SEEK_POSITION_HI, REG_SEEK_POSITION_LO, adc_get_position_value());
178  registers_write_word(REG_SEEK_VELOCITY_HI, REG_SEEK_VELOCITY_LO, 0);
179 
180  // XXX Enable PWM and writing. I do this for now to make development and
181  // XXX tuning a bit easier. Constantly manually setting these values to
182  // XXX turn the servo on and write the gain values get's to be a pain.
183  pwm_enable();
184  registers_write_enable();
185 
186  // This is the main processing loop for the servo. It basically looks
187  // for new position, power or TWI commands to be processed.
188 
189  for (;;)
190  {
191 
193  RS485CMD();
194 
195 
196 
197 
198 
199  // Is position value ready?
200  if (adc_position_value_is_ready())
201  {
202  int16_t pwm;
203  int16_t position;
204 
205 #if PULSE_CONTROL_ENABLED
206  // Give pulse control a chance to update the seek position.
207  pulse_control_update();
208 #endif
209 
210 #if CURVE_MOTION_ENABLED
211  // Give the motion curve a chance to update the seek position and velocity.
212  motion_next(10);
213 #endif
214 
215  // Get the new position value.
216  position = (int16_t) adc_get_position_value();
217 
218 
219 #if ESTIMATOR_ENABLED
220  // Estimate velocity.
221  estimate_velocity(position);
222 #endif
223 
224 #if PID_MOTION_ENABLED
225  // Call the PID algorithm module to get a new PWM value.
226  pwm = pid_position_to_pwm(position);
227 #endif
228 
229 #if IPD_MOTION_ENABLED
230  // Call the IPD algorithm module to get a new PWM value.
231  pwm = ipd_position_to_pwm(position);
232 #endif
233 
234 #if REGULATOR_MOTION_ENABLED
235  // Call the state regulator algorithm module to get a new PWM value.
236  pwm = regulator_position_to_pwm(position);
237 #endif
238 
239  // Update the servo movement as indicated by the PWM value.
240  // Sanity checks are performed against the position value.
241  pwm_update(position, pwm);
242  }
243 
244  // Is a power value ready?
245  if (adc_power_value_is_ready())
246  {
247  // Get the new power value.
248  uint16_t power = adc_get_power_value();
249 
250  // Update the power value for reporting.
251  power_update(power);
252  }
253 
254  //UartPutChar('#');
255  // // Was a command recieved?
256  // if (twi_data_in_receive_buffer())
257  // {
258  // // Handle any TWI command.
259  // handle_twi_command();
260  // }
261 
262 #if MAIN_MOTION_TEST_ENABLED
263  // This code is in place for having the servo drive itself between
264  // two positions to aid in the servo tuning process. This code
265  // should normally be disabled in config.h.
266 #if CURVE_MOTION_ENABLED
267  if (motion_time_left() == 0)
268  {
269  registers_write_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO, 2000);
270  registers_write_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO, 0x0100);
271  motion_append();
272  registers_write_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO, 1000);
273  registers_write_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO, 0x0300);
274  motion_append();
275  registers_write_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO, 2000);
276  registers_write_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO, 0x0300);
277  motion_append();
278  registers_write_word(REG_CURVE_DELTA_HI, REG_CURVE_DELTA_LO, 1000);
279  registers_write_word(REG_CURVE_POSITION_HI, REG_CURVE_POSITION_LO, 0x0100);
280  motion_append();
281  }
282 #else
283  {
284  // Get the timer.
285  uint16_t timer = timer_get();
286 
287  // Reset the timer if greater than 800.
288  if (timer > 800) timer_set(0);
289 
290  // Look for specific events.
291  if (timer == 0)
292  {
293  registers_write_word(REG_SEEK_HI, REG_SEEK_LO, 0x0100);
294  }
295  else if (timer == 400)
296  {
297  registers_write_word(REG_SEEK_HI, REG_SEEK_LO, 0x0300);
298  }
299  }
300 #endif
301 #endif
302  }
303 
304  return 0;
