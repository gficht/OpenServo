198 // Read a single byte from the registers.
199 inline static uint8_t registers_read_byte(uint8_t address)
200 {
201  return registers[address];
202 }
203

207 {
208  registers[address] = value;
209 }

212 inline static void registers_write_enable(void)
213 {
214  uint8_t flags_lo = registers_read_byte(REG_FLAGS_LO);
215 
216  // Enable PWM to the servo motor.
217  registers_write_byte(REG_FLAGS_LO, flags_lo | (1<<FLAGS_LO_WRITE_ENABLED));
218 }
219 
220 
221 inline static void registers_write_disable(void)
222 {
223  uint8_t flags_lo = registers_read_byte(REG_FLAGS_LO);
224 
225  // Disable PWM to the servo motor.
226  registers_write_byte(REG_FLAGS_LO, flags_lo & ~(1<<FLAGS_LO_WRITE_ENABLED));
227 }
228 
229 
230 inline static uint8_t registers_is_write_enabled(void)
231 {
232  return (registers_read_byte(REG_FLAGS_LO) & (1<<FLAGS_LO_WRITE_ENABLED)) ? 1 : 0;
233 }
234 
235 
236 inline static uint8_t registers_is_write_disabled(void)
237 {
238  return (registers_read_byte(REG_FLAGS_LO) & (1<<FLAGS_LO_WRITE_ENABLED)) ? 0 : 1;
239 }
