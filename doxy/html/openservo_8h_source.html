61 // Disable interrupts and returns SREG value used to restore interrupts.
62 inline static uint8_t disable_interrupts(void)
63 {
64  uint8_t sreg;
65 
66  __asm__ volatile (
67  "in %0,__SREG__\n\t"
68  "cli\n\t"
69  : "=r" ((uint8_t) sreg)
70  :
71  );
72 
73  return sreg;
74 }
75 
76 // Restore interrupts Enables interrupts according to the SREG.
77 inline static void restore_interrupts(uint8_t sreg)
78 {
79  __asm__ volatile (
80  "out __SREG__,%0\n\t"
81  :
82  : "r" ((uint8_t) sreg)
83  );
84 }
