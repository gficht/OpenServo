uint8_t sreg;
uint16_t value;
// Clear interrupts.
__asm__ volatile ("in %0,__SREG__\n\tcli\n\t" : "=&r" (sreg));
// Read the registers.
value = (registers[address_hi] << 8) | registers[address_lo];
// Restore status.
__asm__ volatile ("out __SREG__,%0\n\t" : : "r" (sreg));
return value;
}
