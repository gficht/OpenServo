138 /*
139  Smart version of movw:
140  - uses "movw" if possible (supported by MCU, and both registers even)
141  - handles overlapping register pairs correctly
142  - no instruction generated if source and destination are the same
143  (may expand to 0, 1 or 2 instructions).
144  */
145 
146 .macro X_movw dst src
147  .L_movw_dst = -1
148  .L_movw_src = -1
149  .L_movw_n = 0
150  .irp reg, r0, r1, r2, r3, r4, r5, r6, r7, r8, r9, \
151  r10,r11,r12,r13,r14,r15,r16,r17,r18,r19, \
152  r20,r21,r22,r23,r24,r25,r26,r27,r28,r29, \
153  r30,r31
154  .ifc \reg,\dst
155  .L_movw_dst = .L_movw_n
156  .endif
157  .ifc \reg,\src
158  .L_movw_src = .L_movw_n
159  .endif
160  .L_movw_n = .L_movw_n + 1
161  .endr
162  .L_movw_n = 0
163  .irp reg, R0, R1, R2, R3, R4, R5, R6, R7, R8, R9, \
164  R10,R11,R12,R13,R14,R15,R16,R17,R18,R19, \
165  R20,R21,R22,R23,R24,R25,R26,R27,R28,R29, \
166  R30,R31
167  .ifc \reg,\dst
168  .L_movw_dst = .L_movw_n
169  .endif
170  .ifc \reg,\src
171  .L_movw_src = .L_movw_n
172  .endif
173  .L_movw_n = .L_movw_n + 1
174  .endr
175  .if .L_movw_dst < 0
176  .L_movw_n = 0
177  .rept 32
178  .if \dst == .L_movw_n
179  .L_movw_dst = .L_movw_n
180  .endif
181  .L_movw_n = .L_movw_n + 1
182  .endr
183  .endif
184  .if .L_movw_src < 0
185  .L_movw_n = 0
186  .rept 32
187  .if \src == .L_movw_n
188  .L_movw_src = .L_movw_n
189  .endif
190  .L_movw_n = .L_movw_n + 1
191  .endr
192  .endif
193  .if (.L_movw_dst < 0) || (.L_movw_src < 0)
194  .err ; Invalid 'X_movw' arg.
195  .endif
196 
197  .if ((.L_movw_src) - (.L_movw_dst)) /* different registers */
198  .if (((.L_movw_src) | (.L_movw_dst)) & 0x01)
199  .if (((.L_movw_src)-(.L_movw_dst)) & 0x80) /* src < dest */
200  mov (.L_movw_dst)+1, (.L_movw_src)+1
201  mov (.L_movw_dst), (.L_movw_src)
202  .else /* src > dest */
203  mov (.L_movw_dst), (.L_movw_src)
204  mov (.L_movw_dst)+1, (.L_movw_src)+1
205  .endif
206  .else /* both even -> overlap not possible */
207 #if defined(__AVR_HAVE_MOVW__) && __AVR_HAVE_MOVW__
208  movw \dst, \src
209 #else
210  mov (.L_movw_dst), (.L_movw_src)
211  mov (.L_movw_dst)+1, (.L_movw_src)+1
212 #endif
213  .endif
214  .endif
215 .endm

217 /* Macro 'X_lpm' extends enhanced lpm instruction for classic chips.
218  Usage:
219  X_lpm reg, dst
220  where
221  reg is 0..31, r0..r31 or R0..R31
222  dst is z, Z, z+ or Z+
223  It is possible to omit both arguments.
224 
225  Possible results for classic chips:
226  lpm
227  lpm / mov Rd,r0
228  lpm / adiw ZL,1
229  lpm / mov Rd,r0 / adiw ZL,1
230 
231  For enhanced chips it is one instruction always.
232 
233  ATTENTION: unlike enhanced chips SREG (S,V,N,Z,C) flags are
234  changed in case of 'Z+' dst. R0 is scratch.
235  */
236 .macro X_lpm dst=r0, src=Z
237 
238  /* dst evaluation */
239  .L_lpm_dst = -1
240 
241  .L_lpm_n = 0
242  .irp reg, r0, r1, r2, r3, r4, r5, r6, r7, r8, r9, \
243  r10,r11,r12,r13,r14,r15,r16,r17,r18,r19, \
244  r20,r21,r22,r23,r24,r25,r26,r27,r28,r29, \
245  r30,r31
246  .ifc \reg,\dst
247  .L_lpm_dst = .L_lpm_n
248  .endif
249  .L_lpm_n = .L_lpm_n + 1
250  .endr
251 
252  .L_lpm_n = 0
253  .irp reg, R0, R1, R2, R3, R4, R5, R6, R7, R8, R9, \
254  R10,R11,R12,R13,R14,R15,R16,R17,R18,R19, \
255  R20,R21,R22,R23,R24,R25,R26,R27,R28,R29, \
256  R30,R31
257  .ifc \reg,\dst
258  .L_lpm_dst = .L_lpm_n
259  .endif
260  .L_lpm_n = .L_lpm_n + 1
261  .endr
262 
263  .if .L_lpm_dst < 0
264  .L_lpm_n = 0
265  .rept 32
266  .if \dst == .L_lpm_n
267  .L_lpm_dst = .L_lpm_n
268  .endif
269  .L_lpm_n = .L_lpm_n + 1
270  .endr
271  .endif
272 
273  .if (.L_lpm_dst < 0)
274  .err ; Invalid dst arg of 'X_lpm' macro.
275  .endif
276 
277  /* src evaluation */
278  .L_lpm_src = -1
279  .L_lpm_n = 0
280  .irp reg, z,Z,z+,Z+
281  .ifc \reg,\src
282  .L_lpm_src = .L_lpm_n
283  .endif
284  .L_lpm_n = .L_lpm_n + 1
285  .endr
286 
287  .if (.L_lpm_src < 0)
288  .err ; Invalid src arg of 'X_lpm' macro.
289  .endif
290 
291  /* instruction(s) */
292  .if .L_lpm_src < 2
293  .if .L_lpm_dst == 0
294  lpm
295  .else
296 #if defined(__AVR_HAVE_LPMX__) && __AVR_HAVE_LPMX__
297  lpm .L_lpm_dst, Z
298 #else
299  lpm
300  mov .L_lpm_dst, r0
301 #endif
302  .endif
303  .else
304  .if (.L_lpm_dst >= 30)
305  .err ; Registers 30 and 31 are inhibited as 'X_lpm *,Z+' dst.
306  .endif
307 #if defined(__AVR_HAVE_LPMX__) && __AVR_HAVE_LPMX__
308  lpm .L_lpm_dst, Z+
309 #else
310  lpm
311  .if .L_lpm_dst
312  mov .L_lpm_dst, r0
313  .endif
314  adiw r30, 1
315 #endif
316  .endif
317 .endm

319 /*
320  LPM_R0_ZPLUS_INIT is used before the loop to initialize RAMPZ
321  for future devices with RAMPZ:Z auto-increment - [e]lpm r0, Z+.
322 
323  LPM_R0_ZPLUS_NEXT is used inside the loop to load a byte from
324  the program memory at [RAMPZ:]Z to R0, and increment [RAMPZ:]Z.
325 
326  The argument in both macros is a register that contains the
327  high byte (bits 23-16) of the address, bits 15-0 should be in
328  the Z (r31:r30) register. It can be any register except for:
329  r0, r1 (__zero_reg__ - assumed to always contain 0), r30, r31.
330  */
331 
332  .macro LPM_R0_ZPLUS_INIT hhi
333 #if __AVR_ENHANCED__
334  #if BIG_CODE
335  out _SFR_IO_ADDR(RAMPZ), \hhi
336  #endif
337 #endif
338  .endm
339 
340  .macro LPM_R0_ZPLUS_NEXT hhi
341 #if __AVR_ENHANCED__
342  #if BIG_CODE
343  /* ELPM with RAMPZ:Z post-increment, load RAMPZ only once */
344  elpm r0, Z+
345  #else
346  /* LPM with Z post-increment, max 64K, no RAMPZ (ATmega83/161/163/32) */
347  lpm r0, Z+
348  #endif
349 #else
350  #if BIG_CODE
351  /* ELPM without post-increment, load RAMPZ each time (ATmega103) */
352  out _SFR_IO_ADDR(RAMPZ), \hhi
353  elpm
354  adiw r30,1
355  adc \hhi, __zero_reg__
356  #else
357  /* LPM without post-increment, max 64K, no RAMPZ (AT90S*) */
358  lpm
359  adiw r30,1
360  #endif
361 #endif
362  .endm
