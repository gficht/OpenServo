54 inline static void motion_enable(void)
55 {
56  uint8_t flags_lo = registers_read_byte(REG_FLAGS_LO);
57 
58  // Enable PWM to the servo motor.
59  registers_write_byte(REG_FLAGS_LO, flags_lo | (1<<FLAGS_LO_MOTION_ENABLED));
60 }
61 
62 
63 inline static void motion_disable(void)
64 {
65  uint8_t flags_lo = registers_read_byte(REG_FLAGS_LO);
66 
67  // Disable PWM to the servo motor.
68  registers_write_byte(REG_FLAGS_LO, flags_lo & ~(1<<FLAGS_LO_MOTION_ENABLED));
69 }

72 inline static uint32_t motion_time_left(void)
73 // Get the remaining time of the buffered curves.
74 {
75  // The time left is the buffer duration minus the buffer counter.
76  return motion_duration - motion_counter;
77 }
